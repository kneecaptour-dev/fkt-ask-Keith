  <script>
  const key = 'AIzaSyCzUozYqlU_k_QP5JQ0Qv09Fy1036PHF6Q';
const url = `https://www.googleapis.com/youtube/v3/search?part=snippet&q=easy+recipes&type=video&maxResults=10&key={key}`;

    fetch(url)
      .then(r => r.json())
      .then(data => {
        const grid = document.getElementById('grid');
        grid.innerHTML = '';
        data.items.forEach(item => {
          const vid = item.id.videoId;
          const title = item.snippet.title;
          const thumb = item.snippet.thumbnails.medium.url;
          grid.insertAdjacentHTML('beforeend', `
            <div class="card" onclick="play('${vid}')">
              <img src="${thumb}" alt="thumb">
              <p>${title}</p>
              <a class="buy" href="https://example.com" target="_blank">Get Recipe Kit</a>
            </div>
          `);
        });
      })
      .catch(err => console.error(err));

    function play(id){
      document.getElementById('player').src = `https://www.youtube.com/embed/${id}?autoplay=1`;
      document.getElementById('overlay').style.display = 'flex';
    }
    function closeOverlay(){
      document.getElementById('player').src = '';
      document.getElementById('overlay').style.display = 'none';
    }
  </script>
